🔄 自主巡检系统
实现自动巡逻路径规划、状态监控、异常检测与报警
📄 涉及文件
💡 核心知识点
from nav2_simple_commander.robot_navigator import BasicNavigator import rclpy from rclpy.node import Node class PatrolNode(Node): def __init__(self): super().__init__('patrol_node') self.navigator = BasicNavigator() def patrol(self, points): for i, point in enumerate(points): self.get_logger().info(f'导航到点 {i+1}/{len(points)}') self.navigator.goToPose(point) while not self.navigator.isTaskComplete(): feedback = self.navigator.getFeedback() if feedback: self.get_logger().info(f'剩余距离: {feedback.distance_remaining:.2f}m') rclpy.spin_once(self, timeout_sec=0.5) self.get_logger().info(f'到达点 {i+1}，等待5秒') rclpy.sleep(5)
